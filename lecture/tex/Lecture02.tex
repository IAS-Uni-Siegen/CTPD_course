%% Switching function and phase voltages %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Switching function and phase voltages}
    Per VSI lag we define a \hl{switching function}
    \begin{equation}
        \begin{split}
            s_\mathrm{a}(t) & = \begin{cases}
                                    1,  & \text{if } T_1 \text{ is on and } T_2 \text{ is off}, \\
                                    -1, & \text{if } T_1 \text{ is off and } T_2 \text{ is on},
                                \end{cases} \\
            s_\mathrm{b}(t) & = \begin{cases}
                                    1,  & \text{if } T_3 \text{ is on and } T_4 \text{ is off}, \\
                                    -1, & \text{if } T_3 \text{ is off and } T_4 \text{ is on},
                                \end{cases} \\
            s_\mathrm{c}(t) & = \begin{cases}
                                    1,  & \text{if } T_5 \text{ is on and } T_6 \text{ is off}, \\
                                    -1, & \text{if } T_5 \text{ is off and } T_6 \text{ is on}.
                                \end{cases}
        \end{split}
        \label{eq:2L_VSI_switching_functions}
    \end{equation}
    From this we receive the \hl{VSI phase voltages} (with lower index $\mathrm{i}$ for inverter output) against reference as
    \begin{equation}
        u_\mathrm{ai}(t) = \frac{u_\mathrm{dc}(t)}{2} s_\mathrm{a}(t), \quad
        u_\mathrm{bi}(t) = \frac{u_\mathrm{dc}(t)}{2} s_\mathrm{b}(t), \quad
        u_\mathrm{ci}(t) = \frac{u_\mathrm{dc}(t)}{2} s_\mathrm{c}(t)
        \label{eq:VSI_phase_voltages}
    \end{equation}
    and the \hl{DC-link current} is $i_\mathrm{dc}(t) = \nicefrac{1}{2}\sum_{k=\mathrm{a, b, c}}s_k(t)i_k(t)$.
\end{frame}

\begin{frame}
    \frametitle{Three-phase converter with symmetric load in star connection (cont.)}
    Assuming a \hl{star-connected load} and that the star point is not connected to ground, $u_{\mathrm{N}}(t)\neq 0$ may occur leading to a load voltage of
    \begin{equation}
        u_{\mathrm{a}}(t) = u_{\mathrm{ai}}(t) - u_{\mathrm{N}}(t), \quad u_{\mathrm{b}}(t) = u_{\mathrm{bi}}(t) - u_{\mathrm{N}}(t), \quad u_{\mathrm{c}}(t) = u_{\mathrm{ci}}(t) - u_{\mathrm{N}}(t).
    \end{equation}\pause
    To calculate $u_{\mathrm{N}}(t)$ one can utilize the load equation (assuming an \hl{inductive load} as approximation of an electrical machine):
    \begin{equation}
        u_{i}(t) = L \frac{\mathrm{d}}{\mathrm{d}t} i_{i}(t) + u_{\mathrm{N}}(t), \quad i\in \{\mathrm{a, b, c}\}
    \end{equation}\pause
    summing up to
    \begin{equation}
        3u_{\mathrm{N}}(t) +  L \frac{\mathrm{d}}{\mathrm{d}t} \underbrace{\left(i_{\mathrm{a}}(t) + i_{\mathrm{b}}(t) + i_{\mathrm{c}}(t)\right)}_{=0} = u_{\mathrm{ai}}(t) + u_{\mathrm{bi}}(t) +u_{\mathrm{ci}}(t)
    \end{equation}\pause
    and finally delivering the \hl{star-to-ground voltage} as
    \begin{equation}
        u_{\mathrm{N}}(t) = \frac{1}{3} \left(u_{\mathrm{ai}}(t) + u_{\mathrm{bi}}(t) +u_{\mathrm{ci}}(t)\right).
    \end{equation}
\end{frame}

\begin{frame}
    \frametitle{Elementary voltage vectors}
    Based on \eqref{eq:2L_VSI_switching_functions} and \eqref{eq:VSI_phase_voltages} we define the \hl{elementary voltage vectors} as
    \begin{equation}
        \bm{u}_\mathrm{i}(t) = \frac{u_\mathrm{dc}(t)}{2} \begin{bmatrix}
            s_\mathrm{a}(t) \\
            s_\mathrm{b}(t) \\
            s_\mathrm{c}(t)
        \end{bmatrix}\in \left\{\bm{u}_{\mathrm{i}0}(t), \bm{u}_{\mathrm{i}1}(t), \ldots, \bm{u}_{\mathrm{i}7}(t) \right\}.
    \end{equation}
    Given the discrete nature of the switching functions, \hl{only $2^3=8$ different voltage vectors} can be generated by the three-phase 2L-VSI  with $\bm{u}_{\mathrm{i}k}(t)$ being the voltage vector corresponding to the switching state $k$ as defined in \tabref{tab:2L_VSI_Switching_states_resulting_voltages_star_connection}.\pause  Applying the \hl{simplified Clarke transformation} \eqref{eq:Clarke_transformation_zero_component_free} to the elementary voltage vectors, we can represent them in the $\upalpha\upbeta$-plane:
    \begin{equation}
        \bm{u}_\mathrm{\upalpha\upbeta i}(t) = \begin{bmatrix}
            u_\mathrm{\upalpha i}(t) \\
            u_\mathrm{\upbeta i}(t)
        \end{bmatrix}= \bm{T}_{23}\bm{u}_\mathrm{i}(t).
    \end{equation}
    Here, we \hl{neglect the voltage zero-component} as the star connection condition \eqref{eq:star_connection_conditions} prevents the generation of a zero-component current (not considering \hl{parasitic effects}, cf. \figref{fig:VSI_three_phase_two_level_bridge_converter_load_star_parasitic_currents}).
\end{frame}

\begin{frame}
    \frametitle{PWM approximation error}
    \begin{columns}
        \begin{column}{0.49\textwidth}
            To evaluate the error between the reference $s^*(t)$ and the switched output voltage $u(t)$, we introduce the following \hl{normalized integral difference}:
            \begin{equation}
                e(t) = \frac{1}{T_\mathrm{s}}\int_{t_0}^{t}\left(s^*(\tau)-s(\tau)\right)\mathrm{d}\tau.
            \end{equation}\pause
            This error can be interpreted as the resulting \hl{current ripple} assuming a pure inductive load $L$ at a constant input voltage $u_\mathrm{dc}(t)=u_\mathrm{dc}$:
            \begin{equation}
                \Delta i(t) = \frac{T_\mathrm{s}u_\mathrm{dc}}{2 L } \left|e(t)\right|.
                \label{eq:current_ripple_PWM_single_phase_ACDC}
            \end{equation}

\begin{frame}
    \frametitle{Overmodulation (cont.)}
    \begin{columns}
        \begin{column}{0.45\textwidth}
            Considering a normalized sinusoidal input reference
            $$ s^*(t) = m^*\sin(\omega t)=\frac{\hat{u}^*}{\nicefrac{u_\mathrm{dc}}{2}}\sin(\omega t)$$
            with the \hl{modulation ratio $m^*$} one can distinguish two PWM operation areas:
            \begin{itemize}
                \item $m* \leq 1$: linear modulation,
                \item $m* > 1$: overmodulation.
            \end{itemize}
            \vspace{-0.25cm}
            \onslide<2->{%
                \begin{varblock}{Harmonics}
                    Increasing the modulation ratio beyond unity leads to increased PWM deviations and thus increased harmonics.
                \end{varblock}}
        \end{column}

\begin{frame}
    \frametitle{Inverter voltage utilization for naive three-phase PWM}
    Assume linear modulation for all three phases, i.e.,
    $$s_\mathrm{a}^*(t) = m^* \sin\left(\omega t\right), \quad s_\mathrm{b}^*(t) = m^* \sin\left(\omega t - \frac{2\pi}{3}\right), \quad s_\mathrm{c}^*(t) = m^* \sin\left(\omega t - \frac{4\pi}{3}\right), \quad \text{with} \quad m^*\leq1 .$$ Hence, we have
    \begin{equation}
        \left\{|s^*_\mathrm{a}|, |s^*_\mathrm{b}|, |s^*_\mathrm{c}|\right\} \leq 1.
        \label{eq:linear_mod_per_phase_max}
    \end{equation}

        s^*_\mathrm{a} = s^*_\upalpha, \quad s^*_\mathrm{b} = -\frac{1}{2}s^*_\upalpha + \frac{\sqrt{3}}{2}s^*_\upbeta, \quad s^*_\mathrm{c} = -\frac{1}{2}s^*_\upalpha - \frac{\sqrt{3}}{2}s^*_\upbeta.
        \label{eq:average_switching_values_abc_alpha_beta}
    \end{equation}

    \begin{equation}
        \left|s^*_\upalpha\vphantom{-\frac{1}{2}s^*_\upalpha + \frac{\sqrt{3}}{2}s^*_\upbeta}\right|\leq 1, \quad \left|-\frac{1}{2}s^*_\upalpha + \frac{\sqrt{3}}{2}s^*_\upbeta\right| \leq 1, \quad \left|-\frac{1}{2}s^*_\upalpha - \frac{\sqrt{3}}{2}s^*_\upbeta\right| \leq 1.
        \label{eq:maximum_achievable_alpha_beta_voltage}
    \end{equation}
\end{frame}

\subsection{Zero-sequence injection}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Three-phase PWM with zero-sequence injection %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Three-phase PWM with zero-sequence injection}
    Subtract a zero-sequence component from all phase references
    \begin{equation}
        u_\mathrm{a}^{**}(t) = u_\mathrm{a}^*(t) - u_0(t), \quad u_\mathrm{b}^{**}(t) = u_\mathrm{b}^*(t) - u_0(t), \quad u_\mathrm{c}^{**}(t) = u_\mathrm{c}^*(t) - u_0(t)
    \end{equation}
    does not change the line-to-line voltages
    \begin{equation}
        u_\mathrm{a\text{-}b}^{**}(t) = u_\mathrm{a}^{**}(t) - u_\mathrm{b}^{**}(t) = u_\mathrm{a}^*(t) - u_\mathrm{b}^*(t) = u_\mathrm{a\text{-}b}^*(t), \quad \text{etc.}
    \end{equation}
    and hence does not change the $\upalpha\upbeta$-reference voltage, compare \eqref{eq:Clarke_transformation_zero_component_free}. Consequently, the \hl{zero-sequence injection is not a disturbance from a control perspective}. To find $u_0$, we formulate the following optimization problem (dropping the time argument for better readability):
    \begin{equation}
        \min_{u_0} \max \left\{|u^*_\mathrm{a}-u_0|, |u^*_\mathrm{b}-u_0|, |u^*_\mathrm{c}-u_0|\right\},
        \label{eq:opt_problem_zero_sequence_PWM}
    \end{equation}
    i.e., we want to \hl{find the smallest possible zero-sequence component that is required to meet the highest phase voltage demand}.
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Three-phase PWM with zero-sequence injection (cont.)%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Three-phase PWM with zero-sequence injection (cont.)}
    The problem \eqref{eq:opt_problem_zero_sequence_PWM} is equivalent to
    \begin{equation}
        \min_{u_0, r} r \quad \text{subject to} \quad |u^*_\mathrm{a}-u_0| \leq r, \; |u^*_\mathrm{b}-u_0| \leq r, \; |u^*_\mathrm{c}-u_0| \leq r
    \end{equation}
    with $r\in\mathbb{R}\geq 0$ representing the maximum deviation of the zero-sequence injected phase voltages from the original reference voltages. Rewriting the constraints yields
    \begin{equation}
        u_\mathrm{a}^* - r \leq u_0 \leq u_\mathrm{a}^* + r, \quad u_\mathrm{b}^* - r \leq u_0 \leq u_\mathrm{b}^* + r, \quad u_\mathrm{c}^* - r \leq u_0 \leq u_\mathrm{c}^* + r.
    \end{equation}
    Hence, a feasible $u_0$ exists if and if only the three intervals overlap, i.e.,
    \begin{equation}
        \max\left\{u^*_\mathrm{a}-r, u^*_\mathrm{b}-r, u^*_\mathrm{c}-r\right\} \leq \min\left\{u^*_\mathrm{a}+r, u^*_\mathrm{b}+r, u^*_\mathrm{c}+r\right\}.
        \label{eq:feasible_intervals_r_zero_volt_inject}
    \end{equation}
    Define the auxiliary variables
    \begin{equation}
        u_\mathrm{max} = \max\left\{u^*_\mathrm{a}, u^*_\mathrm{b}, u^*_\mathrm{c}\right\}, \quad u_\mathrm{min} = \min\left\{u^*_\mathrm{a}, u^*_\mathrm{b}, u^*_\mathrm{c}\right\}.
        \label{eq:aux_variables_zero_volt_inject}
    \end{equation}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Three-phase PWM with zero-sequence injection (cont.)%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Three-phase PWM with zero-sequence injection (cont.)}
    Combining \eqref{eq:feasible_intervals_r_zero_volt_inject} and \eqref{eq:aux_variables_zero_volt_inject} yields
    \begin{equation}
        u_\mathrm{max} - r \leq u_\mathrm{min} + r \quad \Leftrightarrow \quad r \geq \frac{u_\mathrm{max} - u_\mathrm{min}}{2}.
        \label{eq:zero_volt_inject_r_max_min_range}
    \end{equation}
    Since we search for the smallest possible $r$, we have
    \begin{equation}
        r = \frac{u_\mathrm{max} - u_\mathrm{min}}{2}.
    \end{equation}
    This observation is identical that the optimal solution is given with equality in \eqref{eq:zero_volt_inject_r_max_min_range}, i.e.,
    \begin{equation}
        u_\mathrm{max} - r = u_\mathrm{min} + r \quad \Leftrightarrow \quad u_0 = \frac{u_\mathrm{max} + u_\mathrm{min}}{2}.
    \end{equation}
    Hence, the \hl{optimal zero-sequence component} is:
    \begin{equation}
        u_0(t) = \frac{1}{2}\left(\max\left\{u^*_\mathrm{a}(t), u^*_\mathrm{b}(t), u^*_\mathrm{c}(t)\right\} + \min\left\{u^*_\mathrm{a}(t), u^*_\mathrm{b}(t), u^*_\mathrm{c}(t)\right\}\right).
    \end{equation}
\end{frame}

%% Proof min-max zero-sequence injection voltage capacity %%

\begin{frame}
    \frametitle{Proof min-max zero-sequence injection voltage capacity}
    Using symmetry we will focus on the \hl{first sector} of the hexagon, i.e., $\angle  \bm{s}_{\upalpha\upbeta}\in[\SI{0}{\degree}, \SI{60}{\degree}]$   (cf. \figref{fig:elementary_voltages_alpha_beta}). Any reference voltage in the first sector can be expressed as a linear combination of the two adjacent switching vectors, i.e.,
    \begin{equation}
        \bm{s}_{\upalpha\upbeta} = \frac{\bm{u}_{\upalpha\upbeta}^*}{\nicefrac{u_\mathrm{dc}}{2}} = \tau_1 \bm{s}_{\mathrm{i}1} + \tau_2 \bm{s}_{\mathrm{i}2}, \quad \text{with} \quad \tau_1, \tau_2 \geq 0, \quad \tau_1 + \tau_2 \leq 1.
    \end{equation}
    Here, $\bm{s}_{\mathrm{i}1}$ and $\bm{s}_{\mathrm{i}2}$ are the normalized elementary switching vectors of the first sector, i.e.
    \begin{equation}
        \bm{s}_{\mathrm{i}1} = \begin{bmatrix}
            \frac{4}{3} & 0
        \end{bmatrix}^\top, \quad \bm{s}_{\mathrm{i}2} = \begin{bmatrix}
            \frac{2}{3} & \frac{2}{\sqrt{3}}
        \end{bmatrix}^\top.
    \end{equation}
    Hence, the normalized phase voltage references can be found be the inverse Clarke transformation as
    \begin{equation}
        \bm{s}^*_\mathrm{abc} = \begin{bmatrix}
            1                & 0                       \\
            -\nicefrac{1}{2} & \nicefrac{\sqrt{3}}{2}  \\
            -\nicefrac{1}{2} & -\nicefrac{\sqrt{3}}{2}
        \end{bmatrix} \left(\tau_1 \bm{s}_{\mathrm{i}1} + \tau_2 \bm{s}_{\mathrm{i}2}\right) = \begin{bmatrix}
            \nicefrac{4}{3}\tau_1 + \nicefrac{2}{3}\tau_2  \\
            -\nicefrac{2}{3}\tau_1 + \nicefrac{2}{3}\tau_2 \\
            -\nicefrac{2}{3}\tau_1 - \nicefrac{4}{3}\tau_2
        \end{bmatrix}.
        \label{eq:first_sector_hexagon_lin_comb_abc}
    \end{equation}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Proof min-max zero-sequence injection voltage capacity (cont.)%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Proof min-max zero-sequence injection voltage capacity (cont.)}
    With $\tau_1, \tau_2 \geq 0$ it follows directly from \eqref{eq:first_sector_hexagon_lin_comb_abc} that
    \begin{equation}
        s_\mathrm{a}^* \geq s_\mathrm{b}^* \geq s_\mathrm{c}^* \quad \Rightarrow \quad  \max\left\{s_\mathrm{a}^*, s_\mathrm{b}^*, s_\mathrm{c}^*\right\} = s_\mathrm{a}^*, \quad \min\left\{s_\mathrm{a}^*, s_\mathrm{b}^*, s_\mathrm{c}^*\right\} = s_\mathrm{c}^*.
    \end{equation}
    Hence, the zero-sequence injected phase voltage is $s_0 = \nicefrac{1}{2}(s_\mathrm{a}^* + s_\mathrm{c}^*)=\nicefrac{1}{3}(\tau_1 - \tau_2)$ leading to the shifted phase voltage references
    \begin{equation}
        s_\mathrm{a}^{**} = s_\mathrm{a}^* - s_0 = \tau_1 + \tau_2, \quad s_\mathrm{b}^{**} = s_\mathrm{b}^* - s_0 = -\tau_1 + \tau_2, \quad s_\mathrm{c}^{**} = s_\mathrm{c}^* - s_0 = -\tau_1 -\tau_2.
    \end{equation}
    Because of $\tau_1 + \tau_2 \leq 1$ we have
    \begin{equation}
        |s_\mathrm{a}^{**}| \leq 1, \quad |s_\mathrm{b}^{**}| \leq 1, \quad |s_\mathrm{c}^{**}| \leq 1,
    \end{equation}
    i.e., \hl{each phase voltage reference stays within the linear modulation range and is realizable}. The transfer to other sectors of the hexagon is straightforward by symmetry.
\end{frame}

\subsection{Overmodulation}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Overmodulation for three-phase PWM with min-max injection %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}
    \frametitle{Overmodulation for three-phase PWM with min-max injection}
    \begin{columns}[onlytextwidth]
        \begin{column}{0.53\textwidth}
            Beyond $m^* > \nicefrac{2}{\sqrt{3}}$ the phase voltage references leave the linear modulation range and are no longer realizable. The resulting overmodulation strategy is a simple \hl{clipping}, i.e.,
            \begin{equation}
                \begin{split}
                    s_{i}^{**}(t)          & = \operatorname{clip}\!\left(s_{i}^{*}(t)-s_0(t)\right), \\
                    \operatorname{clip}(x) & = \min\!\left\{1,\max\!\left\{-1,x\right\}\right\}.
                \end{split}
            \end{equation}
            \begin{varblock}{Control disturbance}
                The clipping renders itself a control disturbance and reduces the degrees of freedom: in the linear range, amplitude and phase of the voltage reference can be controlled independently, while in the overmodulation range those two are coupled.
            \end{varblock}
        \end{column}
        \begin{column}{0.45\textwidth}
            \vspace{-0.2cm}
            \begin{figure}
                \begin{tikzpicture}[x=1.2cm,y=1.2cm,>=Latex]

                    \pgfmathsetmacro{\mStar}{1.24}

%% Min-max injection clipping law %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Min-max injection clipping law in the first hexagon sector}
    Assume the balanced sinusoidal reference vector
    \begin{equation}
        s_\mathrm{a}^*(\theta)=m^*\cos\theta, \qquad
        s_\mathrm{b}^*(\theta)=m^*\cos\!\left(\theta-\frac{2\pi}{3}\right), \qquad
        s_\mathrm{c}^*(\theta)=m^*\cos\!\left(\theta+\frac{2\pi}{3}\right).
        \label{eq:balanced_sinusoidal_reference_PWM}
    \end{equation}
    In the first sector ($0 \le \theta \le \nicefrac{\pi}{3}$), the min-max injected zero-sequence is
    \begin{equation}
        \begin{split}
            s_0(\theta) & = \frac{1}{2}\left(\max\{s_\mathrm{a}^*(\theta), s_\mathrm{b}^*(\theta), s_\mathrm{c}^*(\theta)\} + \min\{s_\mathrm{a}^*(\theta), s_\mathrm{b}^*(\theta), s_\mathrm{c}^*(\theta)\}\right) \\
                        & = \frac{1}{2}\left(s_\mathrm{a}^*(\theta) + s_\mathrm{c}^*(\theta)\right)= \frac{m^*}{2}\cos\!\left(\theta+\frac{\pi}{3}\right), \quad \text{for} \quad 0 \le \theta \le \frac{\pi}{3}.
        \end{split}
    \end{equation}
    Hence, the unclipped injected voltage references for the first sector are
    \begin{equation}
        \begin{split}
            s_\mathrm{a}^*(\theta) - s_0(\theta) & = m^*\cos\theta - \frac{m^*}{2}\cos\!\left(\theta+\frac{\pi}{3}\right) = \frac{\sqrt{3}m^*}{2}\cos\!\left(\theta-\frac{\pi}{6}\right),                                \\
            s_\mathrm{b}^*(\theta) - s_0(\theta) & = m^*\cos\!\left(\theta-\frac{2\pi}{3}\right) - \frac{m^*}{2}\cos\!\left(\theta+\frac{\pi}{3}\right) = \frac{3m^*}{2}\cos\!\left(\theta-\frac{2\pi}{3}\right),        \\
            s_\mathrm{c}^*(\theta) - s_0(\theta) & = m^*\cos\!\left(\theta+\frac{2\pi}{3}\right) - \frac{m^*}{2}\cos\!\left(\theta+\frac{\pi}{3}\right) = \frac{\sqrt{3}m^*}{2}\cos\!\left(\theta+\frac{5\pi}{6}\right).
        \end{split}
        \label{eq:unclipped_ref_overmod_min_max}
    \end{equation}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Min-max injection clipping law %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Min-max injection clipping law in the first hexagon sector (cont.)}
    The cosine terms in \eqref{eq:unclipped_ref_overmod_min_max} have the following maximum absolute values in the first sector:
    \begin{equation}
        \left|\cos\left(\theta - \frac{\pi}{6}\right)\right|  \leq 1,                                                                                     \quad
        \left|\cos\left(\theta - \frac{2\pi}{3}\right)\right|  \leq \frac{1}{2},       \quad
        \left|\cos\left(\theta + \frac{5\pi}{6}\right)\right|  \leq 1, \quad \text{for} \quad 0 \le \theta \le \frac{\pi}{3}.
    \end{equation}
    Since phases $\mathrm{a}$ and $\mathrm{c}$ are symmetric, we can distinguish two cases:
    \begin{equation}
        \max_\theta \left\{|s_\mathrm{a}^*- s_0|, |s_\mathrm{c}^*- s_0|\right\} = \frac{\sqrt{3}}{2}m^* \quad \text{and} \quad \max_\theta |s_\mathrm{b}^*- s_0| = \frac{3}{2}\cdot\frac{1}{2}m^* = \frac{3}{4}m^*.
    \end{equation}
    Hence, in terms of overmodulation clipping we have two regions:
    \begin{itemize}
        \item \textbf{Region I: \(\nicefrac{2}{\sqrt{3}} < m^* \le \nicefrac{4}{3}\)}\\
              Only phases $\mathrm{a}$ and $\mathrm{c}$ are clipped, while phase $\mathrm{b}$ is still realizable without clipping.
        \item \textbf{Region II: \(m^* \ge \nicefrac{4}{3}\)}\\
              All three phases are clipped.
    \end{itemize}
    In order to derive the fundamental voltage amplitude, we need to determine the clipping intervals for those two regions.
\end{frame}

\begin{frame}
    \frametitle{Min-max injection clipping law in the first hexagon sector (region I)}
    In region I, clipping starts and ends at the two first-sector roots of
    \begin{equation}
        \cos\left(\theta - \frac{\pi}{6}\right) = \frac{2}{\sqrt{3}m^*}, \qquad \text{for} \qquad \frac{2}{\sqrt{3}} < m^* \le \frac{4}{3}, \quad 0 \le \theta \le \frac{\pi}{3}.
    \end{equation}
    Since the cosine is symmetric around $\theta = \nicefrac{\pi}{6}$, the clipping interval boundaries are
    \begin{equation}
        \gamma_1 = \frac{\pi}{6} - \arccos\!\left(\frac{2}{\sqrt{3}m^*}\right),
        \qquad
        \gamma_2 = \frac{\pi}{6} + \arccos\!\left(\frac{2}{\sqrt{3}m^*}\right).
    \end{equation}
    Hence, phases $\mathrm{a}$ and $\mathrm{c}$ are clipped for $\theta \in [\gamma_1,\gamma_2]$ with clipping width
    \begin{equation}
        \Delta\gamma = \gamma_2 - \gamma_1 = 2\arccos\!\left(\frac{2}{\sqrt{3}m^*}\right).
    \end{equation}
\end{frame}

%% Min-max injection clipping law in the first hexagon sector (region I, cont.) %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Min-max injection clipping law in the first hexagon sector (region I, cont.)}
    Let
    \begin{equation}
        \underline{s}^{**}(\theta) = s_{\upalpha}^{**}(\theta) + \mathrm{j}s_{\upbeta}^{**}(\theta)
    \end{equation}
    denote the clipped realized normalized voltage vector. Then the positive-sequence fundamental amplitude follows from the first Fourier coefficient as
    \begin{equation}
        m^{(1)} = \left|\frac{1}{2\pi}\int_{0}^{2\pi} \underline{s}^{**}(\theta)e^{-\mathrm{j}\theta}\,\mathrm{d}\theta\right|.
    \end{equation}
    With six-sector symmetry, it is sufficient to integrate over sector 1 only:
    \begin{equation}
        m^{(1)} = \frac{3}{\pi}\int_{0}^{\pi/3} \Re\!\left\{\underline{s}^{**}(\theta)e^{-\mathrm{j}\theta}\right\}\,\mathrm{d}\theta.
    \end{equation}
    Here, $\Re\{\cdot\}$ denotes the real part. In region I, sector 1 splits into $[0,\gamma_1] \cup [\gamma_1,\gamma_2] \cup [\gamma_2,\pi/3]$.
    On the two outer intervals, the realizable vector equals the commanded one,
    \begin{equation}
        \underline{s}^{**}(\theta) = m^*e^{\mathrm{j}\theta}
        \quad \Rightarrow \quad
        \Re\!\left\{\underline{s}^{**}(\theta)e^{-\mathrm{j}\theta}\right\} = m^*.
    \end{equation}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Min-max injection clipping law in the first hexagon sector (region I, cont.) %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Min-max injection clipping law in the first hexagon sector (region I, cont.)}
    Hence, only the clipped interval contributes a nontrivial correction:
    \begin{equation}
        m^{(1)} = \frac{3}{\pi}\left[m^*\left(\frac{\pi}{3}-\Delta\gamma\right) + \int_{\gamma_1}^{\gamma_2} \Re\!\left\{\underline{s}^{**}(\theta)e^{-\mathrm{j}\theta}\right\}\,\mathrm{d}\theta\right].
    \end{equation}
    On the clipped interval we have $s_\mathrm{a}^{**}=1$, $s_\mathrm{c}^{**}=-1$, and with $\psi=\theta-\nicefrac{\pi}{6}$,
    \begin{equation}
        s_\mathrm{b}^{**} = \frac{3m^*}{2}\sin\psi.
    \end{equation}
    Applying the Clarke transformation yields
    \begin{equation}
        s_{\upalpha}^{**}(\theta) = 1 - \frac{m^*}{2}\sin\psi,
        \qquad
        s_{\upbeta}^{**}(\theta) = \frac{1}{\sqrt{3}} + \frac{\sqrt{3}m^*}{2}\sin\psi.
    \end{equation}
    Therefore,
    \begin{equation}
        \Re\!\left\{\underline{s}^{**}(\theta)e^{-\mathrm{j}\theta}\right\}
        = s_{\upalpha}^{**}(\theta)\cos\theta + s_{\upbeta}^{**}(\theta)\sin\theta
        = \frac{2}{\sqrt{3}}\cos\psi + m^*\sin^2\psi.
    \end{equation}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Min-max injection clipping law in the first hexagon sector (region I, cont.) %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Min-max injection clipping law in the first hexagon sector (region I, cont.)}
    Since the clipping interval is symmetric around $\nicefrac{\pi}{6}$ and has width $\Delta\gamma = \gamma_2-\gamma_1$, the shifted variable satisfies $\psi \in [-\nicefrac{\Delta\gamma}{2},\nicefrac{\Delta\gamma}{2}]$, giving
    \begin{equation}
        m^{(1)} = \frac{3}{\pi}\left[m^*\left(\frac{\pi}{3}-\Delta\gamma\right) + \int_{-\nicefrac{\Delta\gamma}{2}}^{\nicefrac{\Delta\gamma}{2}}\left(\frac{2}{\sqrt{3}}\cos\psi + m^*\sin^2\psi\right)\mathrm{d}\psi\right].
    \end{equation}
    After integration, this becomes
    \begin{equation}
        \begin{split}
            m^{(1)} & = \frac{3}{\pi}\left[m^*\frac{\pi}{3} - \frac{m^*}{2}\Delta\gamma - \frac{m^*}{2}\sin\!\left(\Delta\gamma\right) + \frac{4}{\sqrt{3}}\sin\!\left(\frac{\Delta\gamma}{2}\right)\right] \\
                    & = m^* - \frac{3m^*}{2\pi}\Delta\gamma - \frac{3m^*}{2\pi}\sin\!\left(\Delta\gamma\right) + \frac{4\sqrt{3}}{\pi}\sin\!\left(\frac{\Delta\gamma}{2}\right).
        \end{split}
    \end{equation}
    With $\Delta\gamma = 2\arccos\!\left(\frac{2}{\sqrt{3}m^*}\right)$, we receive the final expression for the fundamental modulation amplitude in region I:
    \begin{equation}
        m^{(1)} = m^* - \frac{3m^*}{\pi}\arccos\!\left(\frac{2}{\sqrt{3}m^*}\right) + \frac{2\sqrt{3}}{\pi}\sqrt{1-\frac{4}{3(m^*)^2}}, \quad \text{for} \quad \frac{2}{\sqrt{3}} < m^* \le \frac{4}{3}.
    \end{equation}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Min-max injection clipping law in the first hexagon sector (region II) %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Min-max injection clipping law in the first hexagon sector (region II)}
    In region II, i.e. for $m^* \geq \nicefrac{4}{3}$, phases $\mathrm{a}$ and $\mathrm{c}$ are clipped over the entire first sector.
    Phase $\mathrm{b}$ is unclipped only between the two roots of
    \begin{equation}
        \left|\frac{3m^*}{2}\sin\!\left(\theta-\frac{\pi}{6}\right)\right| = 1,
        \qquad 0 \leq \theta \leq \frac{\pi}{3}.
    \end{equation}
    By symmetry around $\nicefrac{\pi}{6}$, the interval boundaries are
    \begin{equation}
        \delta_1 = \frac{\pi}{6} - \arcsin\!\left(\frac{2}{3m^*}\right),
        \qquad
        \delta_2 = \frac{\pi}{6} + \arcsin\!\left(\frac{2}{3m^*}\right),
    \end{equation}
    with unclipped width
    \begin{equation}
        \Delta\delta = \delta_2 - \delta_1 = 2\arcsin\!\left(\frac{2}{3m^*}\right).
    \end{equation}
    Hence, $s_\mathrm{b}^{**}=-1$ on $[0,\delta_1]$, $s_\mathrm{b}^{**}=+1$ on $[\delta_2,\nicefrac{\pi}{3}]$, and only on $[\delta_1,\delta_2]$ phase $\mathrm{b}$ remains unclipped.
\end{frame}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Min-max injection clipping law in the first hexagon sector (region II, cont.) %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Min-max injection clipping law in the first hexagon sector (region II, cont.)}
    Therefore, the realized space vector is piecewise given by
    \begin{equation}
        \underline{s}^{**}(\theta) =
        \begin{cases}
            \nicefrac{4}{3},                                                & 0 \leq \theta \leq \delta_1,                 \\
            s_{\upalpha}^{**}(\theta) + \mathrm{j}s_{\upbeta}^{**}(\theta), & \delta_1 \leq \theta \leq \delta_2,          \\
            \nicefrac{2}{3} + \mathrm{j}\nicefrac{2}{\sqrt{3}},             & \delta_2 \leq \theta \leq \nicefrac{\pi}{3},
        \end{cases}
    \end{equation}
    where on the middle interval the same edge relation as in region I applies:
    \begin{equation}
        \Re\!\left\{\underline{s}^{**}(\theta)e^{-\mathrm{j}\theta}\right\}
        = \frac{2}{\sqrt{3}}\cos\psi + m^*\sin^2\psi,
        \qquad \psi = \theta - \frac{\pi}{6}.
    \end{equation}
    Hence,
    \begin{equation}
        \begin{split}
            m^{(1)} = \frac{3}{\pi}\Bigg[ & \int_{0}^{\delta_1} \frac{4}{3}\cos\theta\,\mathrm{d}\theta
            + \int_{-\nicefrac{\Delta\delta}{2}}^{\nicefrac{\Delta\delta}{2}}\left(\frac{2}{\sqrt{3}}\cos\psi + m^*\sin^2\psi\right)\mathrm{d}\psi                        \\
                                          & + \int_{\delta_2}^{\nicefrac{\pi}{3}}\left(\frac{2}{3}\cos\theta + \frac{2}{\sqrt{3}}\sin\theta\right)\mathrm{d}\theta\Bigg].
        \end{split}
    \end{equation}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Min-max injection clipping law in the first hexagon sector (region II, cont.) %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Min-max injection clipping law in the first hexagon sector (region II, cont.)}
    Evaluating the integrals and using $\delta_1 = \nicefrac{\pi}{6} - \nicefrac{\Delta\delta}{2}$ gives
    \begin{equation}
        m^{(1)} = \frac{3}{\pi}\left[\frac{4}{3}\cos\!\left(\frac{\Delta\delta}{2}\right) + \frac{m^*}{2}\Delta\delta - \frac{m^*}{2}\sin\!\left(\Delta\delta\right)\right].
    \end{equation}
    Hence, the achievable fundamental modulation amplitude in region II is
    \begin{equation}
        m^{(1)} = \frac{4}{\pi}\cos\!\left(\frac{\Delta\delta}{2}\right) + \frac{3m^*}{2\pi}\Delta\delta - \frac{3m^*}{2\pi}\sin\!\left(\Delta\delta\right).
    \end{equation}
    With $\Delta\delta = 2\arcsin\!\left(\frac{2}{3m^*}\right)$, this becomes
    \begin{equation}
        m^{(1)} = \frac{3m^*}{\pi}\arcsin\!\left(\frac{2}{3m^*}\right) + \frac{2}{\pi}\sqrt{1-\frac{4}{9(m^*)^2}}, \quad \text{for} \quad m^* \geq \frac{4}{3}.
    \end{equation}
    Hence, for the \hl{six-step limit} one obtains
    \begin{equation}
        \lim_{m^*\to\infty} m^{(1)} = \frac{3}{\pi}\lim_{m^*\to\infty}\left(m^*\arcsin\!\left(\frac{2}{3m^*}\right)\right) + \frac{2}{\pi} = \frac{2}{\pi} + \frac{2}{\pi} = \frac{4}{\pi}.
    \end{equation}
\end{frame}

\subsection{Discontinous flat-top pulse width modulation}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Continuous vs. discontinuous PWM %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Continuous vs.\ discontinuous PWM schemes}
    \begin{columns}[onlytextwidth, t]
        \begin{column}{0.49\textwidth}
            \textbf{Continuous PWM}
            \begin{itemize}
                \item All three-phase references are modulated continuously.
                \item Zero-voltage injection is a continuous function of the reference vector, e.g., the min-max injection.
                \item Lower low-order distortion, but higher switching activity.
            \end{itemize}
        \end{column}
        \begin{column}{0.49\textwidth}
            \textbf{Discontinuous PWM}
            \begin{itemize}
                \item The zero-sequence is chosen such that (at least) one phase satisfies
                      \begin{equation}
                          s_p^{**}(t)=\pm 1, \quad p\in\{\mathrm{a,b,c}\},
                      \end{equation}
                      over a finite interval (\hl{clamped leg}).
                \item This reduces switching losses, but changes the harmonic spectrum.
            \end{itemize}
        \end{column}
    \end{columns}
    \begin{block}{Remark}
        Discontinuous PWM (DPWM) schemes offer similar fundamental modulation capability as continuous schemes, but with reduced switching losses at the cost of increased low-order harmonics. There are several DPWM variants which differ in the choice of the clamping angle and the clamped leg with the flat-top modulation, as considered next, being a common scheme.
    \end{block}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Discontinuous flat-top modulation %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Discontinuous flat-top modulation}
    Consider a balanced sinusoidal reference as from \eqref{eq:balanced_sinusoidal_reference_PWM}. Define the phase with the largest absolute instantaneous reference as
    \begin{equation}
        p(\theta)\in \operatorname*{arg\,max}_{i\in\{\mathrm{a,b,c}\}}\left|s_i^*(\theta)\right|.
    \end{equation}
    The \hl{flat-top zero-sequence injection} is then chosen as
    \begin{equation}
        s_0(\theta)=s_{p(\theta)}^*(\theta)-\operatorname{sgn}\!\left(s_{p(\theta)}^*(\theta)\right).
        \label{eq:flat_top_zero_sequence_general}
    \end{equation}
    With
    \begin{equation}
        s^*_{\max}=\max\{s_\mathrm{a}^*,s_\mathrm{b}^*,s_\mathrm{c}^*\}, \qquad
        s^*_{\min}=\min\{s_\mathrm{a}^*,s_\mathrm{b}^*,s_\mathrm{c}^*\},
    \end{equation}
    one may write
    \begin{equation}
        s_0=
        \begin{cases}
            s^*_{\max}-1, & s^*_{\max}\geq -s^*_{\min}, \\
            s^*_{\min}+1, & s^*_{\max}< -s^*_{\min}.
        \end{cases}
        \label{eq:flat_top_zero_sequence_piecewise}
    \end{equation}
    Thus, one phase is clamped to \(+1\) or \(-1\) in each \(60^\circ\) interval, yielding the characteristic \hl{flat-top PWM sequence}.
\end{frame}

%% Flat-top voltage capacity %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Voltage capacity of the flat-top injection}
    Using the first-sector parametrization from \eqref{eq:first_sector_hexagon_lin_comb_abc},
    \begin{equation*}
        \bm{s}^*_\mathrm{abc} =
        \begin{bmatrix}
            \nicefrac{4}{3}\tau_1 + \nicefrac{2}{3}\tau_2  &
            -\nicefrac{2}{3}\tau_1 + \nicefrac{2}{3}\tau_2 &
            -\nicefrac{2}{3}\tau_1 - \nicefrac{4}{3}\tau_2
        \end{bmatrix}^\top,
        \qquad
        \tau_1,\tau_2\geq 0, \quad \tau_1+\tau_2\leq 1,
    \end{equation*}
    the \hl{flat-top PWM definition} clamps the leading positive phase to the upper rail for $\theta\in[-\nicefrac{\pi}{6}, \nicefrac{\pi}{6}]$. Hence, within the upper first sector part, phase \(\mathrm{a}\) is clamped, yielding
    \begin{equation}
        s_0=s_\mathrm{a}^*-1
        \qquad \Rightarrow \qquad
        s_\mathrm{a}^{**}=s_\mathrm{a}^*-s_0=1.
        \label{eq:flat_top_clamping_phase_a}
    \end{equation}
    Consequently, only the modulated phases \(\mathrm{b}\) and \(\mathrm{c}\) must be checked. Inserting \eqref{eq:flat_top_clamping_phase_a} into \eqref{eq:first_sector_hexagon_lin_comb_abc} gives
    \begin{equation}
        s_\mathrm{b}^{**}=1-2\tau_1, \qquad
        s_\mathrm{c}^{**}=1-2(\tau_1+\tau_2).
    \end{equation}
    Because \(0\leq \tau_1\leq 1\) and \(0\leq \tau_1+\tau_2\leq 1\), it follows directly that
    \begin{equation}
        -1\leq s_\mathrm{b}^{**}\leq 1, \qquad -1\leq s_\mathrm{c}^{**}\leq 1.
    \end{equation}
    Thus, every reference in the \hl{first upper sector part of the hexagon} is realizable with flat-top PWM: phase \(\mathrm{a}\) is fixed at \(+1\), while phases \(\mathrm{b}\) and \(\mathrm{c}\) remain within the admissible interval. By symmetry, the \hl{entire voltage hexagon}, including its corners, can be reached.
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Flat-top linear modulation range %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Flat-top linear modulation range}
    In the positive \(\mathrm{a}\)-clamp sector $-\nicefrac{\pi}{6}\leq \theta \leq \nicefrac{\pi}{6}$,  one has \(s_0(\theta)=s_\mathrm{a}^*(\theta)-1\), yielding
    \begin{equation}
        s_\mathrm{a}^{**}(\theta)=1,
        \qquad
        s_\mathrm{b}^{**}(\theta)=1-\sqrt{3}m^*\cos\!\left(\theta+\frac{\pi}{6}\right),
        \qquad
        s_\mathrm{c}^{**}(\theta)=1-\sqrt{3}m^*\cos\!\left(\theta-\frac{\pi}{6}\right).
        \label{eq:flat_top_positive_a_sector}
    \end{equation}
    Since
    \begin{equation}
        \cos\!\left(\theta\pm\frac{\pi}{6}\right)\in\left[\frac{1}{2},1\right]
        \qquad \forall\, \theta\in\left[-\frac{\pi}{6},\frac{\pi}{6}\right],
    \end{equation}
    the following must hold to prevent clipping of a further phase:
    \begin{equation}
        1-\sqrt{3}m^*
        \leq
        s_\mathrm{b}^{**}(\theta),\, s_\mathrm{c}^{**}(\theta)
        \leq
        1-\frac{\sqrt{3}}{2}m^*.
    \end{equation}
    Therefore, requiring the non-clamped phases to stay within the \hl{linear modulation} range yields
    \begin{equation}
        1-\sqrt{3}m^* \geq -1
        \qquad \Leftrightarrow \qquad
        m^*\leq \frac{2}{\sqrt{3}}.
    \end{equation}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Flat-top overmodulation regions %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Flat-top overmodulation regions}
    For the flat-top overmodulation range define
    \begin{equation}
        \gamma=\arccos\!\left(\frac{2}{\sqrt{3}m^*}\right),
        \qquad
        \frac{2}{\sqrt{3}}<m^*\leq \frac{4}{\sqrt{3}}.
        \label{eq:flat_top_gamma_definition}
    \end{equation}
    With \eqref{eq:flat_top_gamma_definition} and \eqref{eq:flat_top_positive_a_sector}, on the positive half-sector \(0\leq \theta\leq \nicefrac{\pi}{6}\), phase \(\mathrm{c}\) is clipped for
    \begin{equation}
        \theta\in\left[\frac{\pi}{6}-\gamma,\frac{\pi}{6}\right].
        \label{eq:region_I_flat_top_theta}
    \end{equation}
    If \(\gamma>\nicefrac{\pi}{6}\), then additionally phase \(\mathrm{b}\) is clipped for
    \begin{equation}
        \theta\in\left[0,\gamma-\frac{\pi}{6}\right].
        \label{eq:region_II_flat_top_theta}
    \end{equation}
    Hence, from inserting \eqref{eq:flat_top_gamma_definition} in \eqref{eq:region_I_flat_top_theta} and \eqref{eq:region_II_flat_top_theta}, the flat-top overmodulation splits into
    \begin{itemize}
        \item \textbf{Region I:} \(\nicefrac{2}{\sqrt{3}}<m^*\leq \nicefrac{4}{3}\),
        \item \textbf{Region II:} \(m^*>  \nicefrac{4}{3}\).
    \end{itemize}
\end{frame}

%% Flat-top region I proof %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Flat-top overmodulation: region I fundamental component}
    Let
    \begin{equation}
        \underline{s}^{**}(\theta)=s_{\upalpha}^{**}(\theta)+\mathrm{j}s_{\upbeta}^{**}(\theta)
    \end{equation}
    denote the realized normalized voltage vector. Using the six clamping sectors and mirror symmetry around each sector center, the positive-sequence fundamental amplitude is
    \begin{equation}
        m^{(1)}=
        \left|
        \frac{1}{2\pi}\int_{0}^{2\pi}\underline{s}^{**}(\theta)e^{-\mathrm{j}\theta}\,\mathrm{d}\theta
        \right|
        =
        \frac{6}{\pi}\int_{0}^{\pi/6}\Re\!\left\{\underline{s}^{**}(\theta)e^{-\mathrm{j}\theta}\right\}\,\mathrm{d}\theta.
        \label{eq:flat_top_fundamental_half_sector}
    \end{equation}
    In \textbf{region I}, define the clipping onset
    \begin{equation}
        \theta_\mathrm{c}=\frac{\pi}{6}-\gamma,
        \qquad
        \frac{2}{\sqrt{3}}<m^*\leq \frac{4}{3}.
    \end{equation}
    Then, on \(0\leq \theta\leq \theta_\mathrm{c}\) the reference is still realizable:
    \begin{equation}
        \underline{s}^{**}(\theta)=m^*e^{\mathrm{j}\theta}
        \qquad \Rightarrow \qquad
        \Re\!\left\{\underline{s}^{**}(\theta)e^{-\mathrm{j}\theta}\right\}=m^*.
    \end{equation}
    On the clipped interval \(\theta_\mathrm{c}\leq \theta\leq \nicefrac{\pi}{6}\), one has
    \begin{equation}
        s_\mathrm{a}^{**}=1, \qquad
        s_\mathrm{c}^{**}=-1, \qquad
        s_\mathrm{b}^{**}=1-\sqrt{3}m^*\cos\!\left(\theta+\frac{\pi}{6}\right).
    \end{equation}
    Applying the Clarke transformation gives
    \begin{equation}
        s_{\upalpha}^{**}(\theta)=\frac{2}{3}+\frac{m^*}{\sqrt{3}}\cos\!\left(\theta+\frac{\pi}{6}\right), \qquad
        s_{\upbeta}^{**}(\theta)=\frac{2}{\sqrt{3}}-m^*\cos\!\left(\theta+\frac{\pi}{6}\right).
    \end{equation}
\end{frame}



%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Flat-top region I proof cont. %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Flat-top overmodulation: region I fundamental component (cont.)}
    Hence, on the clipped interval
    \begin{equation}
        \Re\!\left\{\underline{s}^{**}(\theta)e^{-\mathrm{j}\theta}\right\}
        =
        s_{\upalpha}^{**}(\theta)\cos\theta+s_{\upbeta}^{**}(\theta)\sin\theta
        =
        \frac{2}{3}\cos\theta+\frac{2}{\sqrt{3}}\sin\theta+\frac{m^*}{2}-\frac{\sqrt{3}m^*}{3}\sin(2\theta).
        \label{eq:flat_top_region1_integrand}
    \end{equation}
    Therefore, the region-I fundamental amplitude follows from
    \begin{equation}
        m^{(1)}=
        \frac{6}{\pi}
        \left[
            \int_{0}^{\theta_\mathrm{c}} m^*\,\mathrm{d}\theta
            +
            \int_{\theta_\mathrm{c}}^{\nicefrac{\pi}{6}}
            \left(
            \frac{2}{3}\cos\theta+\frac{2}{\sqrt{3}}\sin\theta+\frac{m^*}{2}-\frac{\sqrt{3}m^*}{3}\sin(2\theta)
            \right)\mathrm{d}\theta
            \right].
        \label{eq:flat_top_region1_integral}
    \end{equation}
    Evaluating the integrals and using \(\cos\gamma=\nicefrac{2}{\sqrt{3}m^*}\) yields
    \begin{equation}
        \begin{split}
            m^{(1)}
             & =
            m^*
            -\frac{3m^*}{\pi}\gamma
            +\frac{\sqrt{3}m^*}{\pi}
            -\frac{4}{\pi}
            +\frac{4}{\sqrt{3}\pi m^*}
            +\frac{2\sqrt{3}}{\pi}\sin\gamma,
            \qquad
            \frac{2}{\sqrt{3}}<m^*\leq \frac{4}{3} \\
             & =
            m^*
            -\frac{3m^*}{\pi}\arccos\!\left(\frac{2}{\sqrt{3}m^*}\right)
            +\frac{\sqrt{3}m^*}{\pi}
            -\frac{4}{\pi}
            +\frac{4}{\sqrt{3}\pi m^*}
            +\frac{2\sqrt{3}}{\pi}\sqrt{1-\frac{4}{3(m^*)^2}}.
        \end{split}
        \label{eq:flat_top_region1_closed_form}
    \end{equation}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Flat-top overmodulation: transition to full six-step %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Flat-top overmodulation: transition to full six-step}
    In \textbf{region II}, i.e. $m^*>\nicefrac{4}{3}$, phase \(\mathrm{c}\) is clipped over the entire half-sector \(0\leq \theta\leq \nicefrac{\pi}{6}\), while phase \(\mathrm{b}\) is clipped additionally for
    \begin{equation}
        0\leq \theta\leq \theta_\mathrm{b},
        \qquad
        \theta_\mathrm{b}=\gamma-\frac{\pi}{6}.
    \end{equation}
    The \hl{full six-step waveform} is reached when phase \(\mathrm{b}\) is also clipped over the complete half-sector, i.e.
    \begin{equation}
        \theta_\mathrm{b}=\frac{\pi}{6}
        \qquad \Leftrightarrow \qquad
        \gamma=\frac{\pi}{3}.
    \end{equation}
    Inserting this in \eqref{eq:flat_top_gamma_definition} gives the command value at which flat-top PWM reaches six-step operation:
    \begin{equation}
        \arccos\!\left(\frac{2}{\sqrt{3}m^*}\right)=\frac{\pi}{3}
        \qquad \Leftrightarrow \qquad
        m^*=\frac{4}{\sqrt{3}}.
    \end{equation}
    Thus, the following derivation applies to the strict region-II interval
    \begin{equation}
        \frac{4}{3}<m^*<\frac{4}{\sqrt{3}}.
    \end{equation}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Flat-top overmodulation: region II fundamental component %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Flat-top overmodulation: region II fundamental component}
    For \(\nicefrac{4}{3}<m^*<\nicefrac{4}{\sqrt{3}}\), the realized vector is piecewise given by
    \begin{equation}
        \underline{s}^{**}(\theta)=
        \begin{cases}
            \nicefrac{4}{3},                                              & 0\leq \theta\leq \theta_\mathrm{b},                 \\[1mm]
            s_{\upalpha}^{**}(\theta)+\mathrm{j}s_{\upbeta}^{**}(\theta), & \theta_\mathrm{b}\leq \theta\leq \nicefrac{\pi}{6},
        \end{cases}
        \label{eq:flat_top_region2_piecewise_vector}
    \end{equation}
    where on the second interval the same expressions as in region I apply, i.e. \eqref{eq:flat_top_region1_integrand}.
    Hence,
    \begin{equation}
        m^{(1)}=
        \frac{6}{\pi}
        \left[
            \int_{0}^{\theta_\mathrm{b}}\frac{4}{3}\cos\theta\,\mathrm{d}\theta
            +
            \int_{\theta_\mathrm{b}}^{\nicefrac{\pi}{6}}
            \left(
            \frac{2}{3}\cos\theta+\frac{2}{\sqrt{3}}\sin\theta+\frac{m^*}{2}-\frac{\sqrt{3}m^*}{3}\sin(2\theta)
            \right)\mathrm{d}\theta
            \right].
        \label{eq:flat_top_region2_integral}
    \end{equation}
    Evaluating \eqref{eq:flat_top_region2_integral} and again using \(\cos\gamma=\nicefrac{2}{\sqrt{3}m^*}\) yields
    \begin{equation}
        \begin{split}
            m^{(1)}
             & =
            m^*
            -\frac{3m^*}{\pi}\gamma
            +\frac{\sqrt{3}m^*}{\pi}
            -\frac{4}{\pi}
            +\frac{4}{\sqrt{3}\pi m^*}
            +\frac{2\sqrt{3}}{\pi}\sin\gamma,
            \qquad
            \frac{4}{3}<m^*<\frac{4}{\sqrt{3}} \\
             & =
            m^*
            -\frac{3m^*}{\pi}\arccos\!\left(\frac{2}{\sqrt{3}m^*}\right)
            +\frac{\sqrt{3}m^*}{\pi}
            -\frac{4}{\pi}
            +\frac{4}{\sqrt{3}\pi m^*}
            +\frac{2\sqrt{3}}{\pi}\sqrt{1-\frac{4}{3(m^*)^2}}.
        \end{split}
        \label{eq:flat_top_region2_closed_form}
    \end{equation}
\end{frame}
