    \begin{itemize}

        \item Understand the trade-offs of different drive control concepts and their suitability for different applications (incl. different machines and converter types).

\begin{frame}
    \frametitle{Modulator-based vs.\ modulator-free inverter actuation}
    \renewcommand{\pro}{\textcolor{signalbeta}{\ding{51}}}%
    \renewcommand{\con}{\textcolor{signaldelta}{\ding{55}}}%
    \begin{table}
        \centering
        \small
        \renewcommand{\arraystretch}{1.3}
        \begin{tabular}{@{}p{3.75cm} c c@{}}
            \toprule
                                  & \textbf{Modulator-based}      & \textbf{Modulator-free}           \\
            \midrule
            Switching frequency   & \pro~Fixed and predictable    & \con~Variable                     \\
            Harmonic spectrum     & \pro~Well-defined             & \con~Spread spectrum              \\
            Dynamic performance   & \con~Slower (modulator delay) & \pro~Faster                       \\
            Implementation effort & \pro~Standard hardware        & \con~Fast DSP/{\textmu}C required \\
            \bottomrule
        \end{tabular}
        \caption{Qualitative comparison of modulator-based and modulator-free inverter actuation}
        \label{tab:modulator_vs_modulatorfree}
    \end{table}
    \vspace{-0.5cm}
    \begin{varblock}{}
        The modulator allows to map continuous control decisions to discrete switching actions. This enables standard linear feedback controllers (i.e., well-understood simple control design) and ensures a predefined switching frequency and harmonic spectrum. However, the modulator delay limits achievable transient control performance.
    \end{varblock}
\end{frame}

\subsection{Single and three-phase alternating current systems}
\label{subsec:single_three_phase_AC_systems}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Single-phase alternating current (AC) system %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Single-phase alternating current (AC) system}
    Consider a general \hl{phase signal} (e.g., voltage, current, flux linkage)
    \begin{equation}
        x(t) = \hat{x}(t)\cos(\varphi_x(t))
    \end{equation}
    with time-varying \hl{amplitude} $\hat{x}(t)\in\mathbb{R}_{\geq 0}$ and \hl{phase} $\varphi_x(t)\in\mathbb{R}$. The \hl{instantaneous angular frequency} of this signal is defined as
    \begin{equation}
        \omega_x(t) = \frac{\mathrm{d}}{\mathrm{d}t}\varphi_x(t).
    \end{equation}
    Assuming that $\varphi_x(t)$ has a \hl{period} $T\in\mathbb{R}_{>0}$, typical (time-varying) signal properties are:
    \begin{itemize}
        \item Average: $\overline{x}(t)=\frac{1}{T}\int_{t-T}^{t} x(\tau) \, \mathrm{d}\tau$,
        \item Rectified average: $\overline{|x|}(t)=\frac{1}{T}\int_{t-T}^{t} |x(\tau)| \, \mathrm{d}\tau$,
        \item Root mean square (RMS): $X(t)=\sqrt{\frac{1}{T}\int_{t-T}^{t} x^2(\tau) \, \mathrm{d}\tau}$.
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Three-phase AC system %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Three-phase AC system}
    Extending the single-phase system to three phases, we have \hl{three-phase signal}
    \begin{equation}
        \bm{x}_\mathrm{abc} = \begin{bmatrix}
            x_\mathrm{a}(t) \\
            x_\mathrm{b}(t) \\
            x_\mathrm{c}(t)
        \end{bmatrix} = \begin{bmatrix}
            \hat{x}_\mathrm{a}(t)\cos(\varphi_{x_\mathrm{a}}(t)) \\
            \hat{x}_\mathrm{b}(t)\cos(\varphi_{x_\mathrm{b}}(t)) \\
            \hat{x}_\mathrm{c}(t)\cos(\varphi_{x_\mathrm{c}}(t))
        \end{bmatrix}.
    \end{equation}
    Here, again $\hat{x}_\mathrm{a}(t), \hat{x}_\mathrm{b}(t), \hat{x}_\mathrm{c}(t)\in\mathbb{R}_{\geq 0}$ are the time-varying amplitudes and $\varphi_{x_\mathrm{a}}(t), \varphi_{x_\mathrm{b}}(t), \varphi_{x_\mathrm{c}}(t)\in\mathbb{R}$ are the time-varying phase angles. Instead of phase signals, we can also define \hl{line-to-line signals} as
    \begin{equation}
        \bm{x}_\mathrm{a\text{-}b\text{-}c} = \begin{bmatrix}
            x_\mathrm{a\text{-}b}(t) \\
            x_\mathrm{b\text{-}c}(t) \\
            x_\mathrm{c\text{-}a}(t)
        \end{bmatrix} = \begin{bmatrix}
            x_\mathrm{a}(t) - x_\mathrm{b}(t) \\
            x_\mathrm{b}(t) - x_\mathrm{c}(t) \\
            x_\mathrm{c}(t) - x_\mathrm{a}(t)
        \end{bmatrix} = \underbrace{\begin{bmatrix}
                1  & -1 & 0  \\
                0  & 1  & -1 \\
                -1 & 0  & 1
            \end{bmatrix}}_{\bm{T}_\Delta} \
        \bm{x}_\mathrm{abc}.
    \end{equation}
    Since $\det(\bm{T}_\Delta) = 0$, we cannot directly transform line-to-line signals back to phase signals (at least not without additional information such as a reference potential against neutral).
\end{frame}

%% Balanced three-phase systems %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Balanced three-phase systems}
    A three-phase system is called \hl{balanced} if the three-phase signals have equal amplitude and are phase-shifted by $120^\circ$ (i.e., $\nicefrac{2\pi}{3}$ radians) with respect to each other. In this case, the three-phase signals can be expressed as
    \begin{equation}
        \begin{split}
            \bm{x}_{\mathrm{abc}}(t) = \begin{bmatrix}
                                           x_\mathrm{a}(t) \\
                                           x_\mathrm{b}(t) \\
                                           x_\mathrm{c}(t)
                                       \end{bmatrix} = \hat{x}(t) \begin{bmatrix}
                                                                      \cos(\varphi_x(t))                             \\
                                                                      \cos\left(\varphi_x(t) - \frac{2\pi}{3}\right) \\
                                                                      \cos\left(\varphi_x(t) - \frac{4\pi}{3}\right)
                                                                  \end{bmatrix}
        \end{split}
    \end{equation}
    with common amplitude and shifted phases
    \begin{equation}
        \begin{gathered}
            \hat{x}(t)    =\hat{x}_\mathrm{a}(t)=\hat{x}_\mathrm{b}(t)=\hat{x}_\mathrm{c}(t),            \\
            \varphi_x(t)  =\varphi_a(t)=\varphi_b(t)+\nicefrac{2\pi}{3}=\varphi_c(t)+\nicefrac{4\pi}{3}.
        \end{gathered}
    \end{equation}
    If these conditions apply, the \hl{zero-sequence component} of the three-phase systems is zero:
    \begin{equation}
        x_0(t) = \frac{1}{3}\left(x_\mathrm{a}(t)+x_\mathrm{b}(t)+x_\mathrm{c}(t)\right) = 0.
    \end{equation}
\end{frame}

%% Clarke transformation %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Clarke transformation}
    Consider any $\bm{x}_\mathrm{abc}(t)\in\mathbb{R}^3$, then the \hl{Clarke transformation} is given by
    \begin{equation}
        \renewcommand{\arraystretch}{1.2}
        \bm{x}_{\upalpha \upbeta \upgamma}(t) = \begin{bmatrix} x_\upalpha(t) \\ x_\upbeta(t) \\ x_\upgamma(t)\end{bmatrix} = \begin{bmatrix}
            \nicefrac{2}{3}        & -\nicefrac{1}{3}       & -\nicefrac{1}{3}        \\
            0                      & \nicefrac{1}{\sqrt{3}} & -\nicefrac{1}{\sqrt{3}} \\
            \nicefrac{\sqrt{2}}{3} & \nicefrac{\sqrt{2}}{3} & \nicefrac{\sqrt{2}}{3}
        \end{bmatrix} \begin{bmatrix} x_\mathrm{a}(t) \\ x_\mathrm{b}(t) \\ x_\mathrm{c}(t)\end{bmatrix} = \bm{T}_\mathrm{c} \bm{x}_\mathrm{abc}.
        \label{eq:Clarke_transformation}
    \end{equation}
    \pause
    Above, $\bm{T}_\mathrm{c}\in\mathbb{R}^{3\times 3}$ is the \hl{Clarke transformation matrix} and $\bm{x}_\mathrm{\upalpha \upbeta \upgamma}\in\mathbb{R}^3$ the transformed vector. The values inside $\bm{T}_\mathrm{c}$ are originating from the angle displacement between the two coordinate system axes (cf. \figref{fig:abc_alphabeta_vectors}):
    \begin{equation}
        \renewcommand{\arraystretch}{1.2}
        \bm{T}_\mathrm{c} = \frac{2}{3}\begin{bmatrix}
            \cos(0)                 & \cos(\nicefrac{2\pi}{3}) & \cos(\nicefrac{4\pi}{3}) \\
            \cos(\nicefrac{\pi}{2}) & \cos(\nicefrac{\pi}{6})  & \cos(\nicefrac{5\pi}{6})
            \\ \nicefrac{1}{\sqrt{2}} & \nicefrac{1}{\sqrt{2}} & \nicefrac{1}{\sqrt{2}}\end{bmatrix} = \begin{bmatrix}
            \nicefrac{2}{3}        & -\nicefrac{1}{3}       & -\nicefrac{1}{3}        \\
            0                      & \nicefrac{1}{\sqrt{3}} & -\nicefrac{1}{\sqrt{3}} \\
            \nicefrac{\sqrt{2}}{3} & \nicefrac{\sqrt{2}}{3} & \nicefrac{\sqrt{2}}{3}
        \end{bmatrix}.
    \end{equation}
    The factor $\nicefrac{2}{3}$ is a normalization factor to preserve the amplitude of the transformed vector, i.e., the amplitude of $\bm{x}_\mathrm{\upalpha \upbeta \upgamma}(t)$ is identical to the amplitude of $\bm{x}_\mathrm{abc}(t)$.
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Clarke transformation (cont.) %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Clarke transformation (cont.)}
    The \hl{inverse transformation} is defined
    \begin{equation}
        \renewcommand{\arraystretch}{1.2}
        \bm{x}_\mathrm{abc}(t)  = \begin{bmatrix}
            1                & 0                       & \nicefrac{1}{\sqrt{2}} \\
            -\nicefrac{1}{2} & \nicefrac{\sqrt{3}}{2}  & \nicefrac{1}{\sqrt{2}} \\
            -\nicefrac{1}{2} & -\nicefrac{\sqrt{3}}{2} & \nicefrac{1}{\sqrt{2}}
        \end{bmatrix} \begin{bmatrix} x_\upalpha(t) \\ x_\upbeta(t) \\ x_\upgamma(t)\end{bmatrix} = \bm{T}_\mathrm{c}^{-1} \bm{x}_{\upalpha \upbeta \upgamma}(t).
    \end{equation}

    \begin{varblock}{$\upgamma$ vs. zero-sequence component}
        The $\upgamma$ component is a scaled version of the zero-sequence component, i.e., $$x_\upgamma(t) = \sqrt{2} x_0(t).$$
        This choice is not unique, i.e., there are Clarke transformation variants which map directly to $x_0(t)$ instead of $x_\upgamma(t)$, but in this case the power-preserving variant from \eqref{eq:Clarke_transformation_power_preserving} is not a simple scalar mapping from \eqref{eq:Clarke_transformation}.
    \end{varblock}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Clarke transformation: amplitude and power scaling %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Clarke transformation: amplitude and power scaling}
    The transformation \eqref{eq:Clarke_transformation} is \hl{amplitude-preserving}, i.e., the amplitude of the $\upalpha\upbeta$ vector is identical to the amplitude of the original abc vector. \pause On the other hand, the power is not preserved, as can be seen from the inner product of the transformed vectors (which commonly occurs in power calculations):
    \begin{equation*}
        \bm{x}_{\mathrm{abc}}\T \bm{y}_{\mathrm{abc}} = \bm{x}_\mathrm{\upalpha \upbeta \upgamma}\T\left(\bm{T}_\mathrm{c}^{-1}\right)\T \bm{T}_\mathrm{c}^{-1}\bm{y}_\mathrm{\upalpha \upbeta \upgamma} \quad \Leftrightarrow \quad x_\mathrm{a}y_\mathrm{a} + x_\mathrm{b}y_\mathrm{b} + x_\mathrm{c}y_\mathrm{c} = \frac{3}{2}\left(x_\upalpha y_\upalpha + x_\upbeta y_\upbeta + x_\upgamma y_\upgamma \right).
    \end{equation*}
    \pause
    The alternative \hl{power-preserving} Clarke transformation variant is given by
    \begin{equation}
        \renewcommand{\arraystretch}{1.2}
        \bm{T}'_\mathrm{c} = \frac{\sqrt{3}}{2} \bm{T}_\mathrm{c} \qquad \left(\bm{T}'_\mathrm{c}\right)^{-1} = \left(\bm{T}'_\mathrm{c}\right)\T,
        \label{eq:Clarke_transformation_power_preserving}
    \end{equation}
    which utilizes an \hl{orthogonal transformation matrix}. \pause However, when using $\bm{T}'_\mathrm{c}$ the amplitude of the transformed vector is not preserved.0\hl{ While being an arbitrary choice, we will stick to \eqref{eq:Clarke_transformation} as a convention for the following.}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Clarke transformation: simplification for zero-component-free vectors %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Clarke transformation: simplification for zero-component-free vectors}
    If the abc vector $\bm{x}_\mathrm{abc}(t)$ is zero-component-free, i.e.,  $$x_\mathrm{a}(t) + x_\mathrm{b}(t) +x_\mathrm{c}(t) = 0,$$
    e.g., the phase currents of a star connected system, \pause the Clarke transformation simplifies to
    \begin{equation}
        \renewcommand{\arraystretch}{1.2}
        \bm{x}_{\upalpha \upbeta}(t) = \begin{bmatrix} x_\upalpha(t) \\ x_\upbeta(t) \end{bmatrix} =\begin{bmatrix}
            \nicefrac{2}{3} & -\nicefrac{1}{3}       & -\nicefrac{1}{3}        \\
            0               & \nicefrac{1}{\sqrt{3}} & -\nicefrac{1}{\sqrt{3}}
        \end{bmatrix} \begin{bmatrix} x_\mathrm{a}(t) \\ x_\mathrm{b}(t) \\ x_\mathrm{c}(t)\end{bmatrix} = \bm{T}_{23} \bm{x}_\mathrm{abc}(t)
        \label{eq:Clarke_transformation_zero_component_free}
    \end{equation}
    \pause
    and
    \begin{equation}
        \renewcommand{\arraystretch}{1.2}
        \bm{x}_\mathrm{abc}(t)  = \begin{bmatrix}
            1                & 0                       \\
            -\nicefrac{1}{2} & \nicefrac{\sqrt{3}}{2}  \\
            -\nicefrac{1}{2} & -\nicefrac{\sqrt{3}}{2}
        \end{bmatrix} \begin{bmatrix} x_\upalpha(t) \\ x_\upbeta(t) \end{bmatrix} = \bm{T}_{32} \bm{x}_{\upalpha \upbeta}(t).
        \label{eq:Clarke_transformation_zero_component_free_inverse}
    \end{equation}
    Here, $\bm{T}_{23}\in\mathbb{R}^{2\times 3}$ and $\bm{T}_{32}\in\mathbb{R}^{3\times 2}$ are the \hl{simplified Clarke transformation matrices} for zero-component-free vectors.
\end{frame}

\begin{frame}
    \frametitle{Clarke transformation: simplification for zero-component-free vectors}
    \begin{figure}

%% Park transformation %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Park transformation}
    The \hl{Park transform rotates} a vector $\bm{x}_{\upalpha\upbeta}\in\mathbb{R}^2$ by a certain angle $\varepsilon$ to obtain $\bm{x}_\mathrm{dq}\in\mathbb{R}^2$, that is,
    \begin{equation}
        \renewcommand{\arraystretch}{1.2}
        \bm{x}_{\mathrm{dq}} = \begin{bmatrix} x_\mathrm{d} \\ x_\mathrm{q} \end{bmatrix} = \begin{bmatrix}
            \cos(\varepsilon)  & \sin(\varepsilon) \\
            -\sin(\varepsilon) & \cos(\varepsilon)
        \end{bmatrix} \begin{bmatrix} x_\upalpha \\ x_\upbeta \end{bmatrix} = \bm{T}_\mathrm{p}^{-1}(\varepsilon) \bm{x}_\mathrm{\upalpha \upbeta}
        \label{eq:Park_transformation}
    \end{equation}
    \pause
    with the counter rotation
    \begin{equation}
        \renewcommand{\arraystretch}{1.2}
        \bm{x}_{\upalpha\upbeta} = \begin{bmatrix}
            \cos(\varepsilon) & -\sin(\varepsilon) \\
            \sin(\varepsilon) & \cos(\varepsilon)
        \end{bmatrix} \begin{bmatrix} x_\mathrm{d} \\ x_\mathrm{q} \end{bmatrix} = \bm{T}_\mathrm{p}(\varepsilon) \bm{x}_{\mathrm{dq}}.
    \end{equation}
    \pause
    Above, $\bm{T}_\mathrm{p}\in\mathbb{R}^{2\times 2}$ is the \hl{Park transformation matrix}. It might be noted that is a (historical) convention to define that $\bm{T}_\mathrm{p}$ rotates into the mathematically positive direction. Depending on the application background and choice of $\varepsilon$ (e.g., alignment with stator / air gap / rotor flux of a certain three-phase machine type), the interpretation of $\bm{x}_\mathrm{dq}$ can vary.
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Park transformation: some properties %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Park transformation: some properties}
    \onslide<+->
    Performing the Park and inverse Park transformation sequentially, does not change the vector:
    \begin{equation}
        \bm{x}_{\upalpha \upbeta} = \bm{T}_\mathrm{p} \bm{T}_\mathrm{p}^{-1} \bm{x}_{\upalpha \upbeta} = \bm{T}_\mathrm{p}^{-1} \bm{T}_\mathrm{p} \bm{x}_{\upalpha \upbeta}.
    \end{equation}
    \onslide<+->
    A frequent rotation within the electric machines and drives context is
    \begin{equation}
        \renewcommand{\arraystretch}{1.1}
        \bm{T}_\mathrm{p}(\varepsilon = \nicefrac{\pi}{2}) = \begin{bmatrix}
            \cos(\nicefrac{\pi}{2}) & -\sin(\nicefrac{\pi}{2}) \\
            \sin(\nicefrac{\pi}{2}) & \cos(\nicefrac{\pi}{2})
        \end{bmatrix} = \begin{bmatrix}
            0 & -1 \\
            1 & 0
        \end{bmatrix} = \bm{J}
    \end{equation}
    leading to the definition of $\bm{J}\in\mathbb{R}^{2 \times 2}$ which will be used for brevity in the following.  \onslide<+-> Moreover, if $\varepsilon$ results from some rotation, i.e., $\nicefrac{\mathrm{d}}{\mathrm{d}t}\,\varepsilon(t)=\omega(t)$, we have:
    \begin{align}
        \renewcommand{\arraystretch}{1.1}
        \uncover<+->{\frac{\mathrm{d}}{\mathrm{d}t}\bm{T}_\mathrm{p}(\varepsilon(t))      & = \begin{bmatrix}
                                                                                                      -\sin(\varepsilon(t)) & -\cos(\varepsilon(t)) \\
                                                                                                      \cos(\varepsilon(t))  & -\sin(\varepsilon(t))
                                                                                                  \end{bmatrix} \frac{\mathrm{d}}{\mathrm{d}t}\varepsilon(t)}  \uncover<+->{= \bm{T}_\mathrm{p}(\varepsilon(t)) \bm{J}\omega(t),} \\
        \uncover<+->{\frac{\mathrm{d}}{\mathrm{d}t}\bm{T}_\mathrm{p}^{-1}(\varepsilon(t)) & = \begin{bmatrix}
                                                                                                      -\sin(\varepsilon(t)) & \cos(\varepsilon(t))  \\
                                                                                                      -\cos(\varepsilon(t)) & -\sin(\varepsilon(t))
                                                                                                  \end{bmatrix} \frac{\mathrm{d}}{\mathrm{d}t}\varepsilon(t)  = -\bm{T}_\mathrm{p}^{-1}(\varepsilon(t)) \bm{J}\omega(t).}
        \label{eq:Park_transformation_derivative}
    \end{align}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Park transformation with zero component %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Park transformation with zero component}
    If a given signal $\bm{x}_{\upalpha\upbeta\upgamma}\in\mathbb{R}^3$ contains a zero component, the Park transformation can be applied to the $\upalpha\upbeta$ components while leaving the $\upgamma$ component unchanged, i.e.,
    \begin{equation}
        \bm{x}_{\mathrm{dq}\upgamma} = \begin{bmatrix} x_\mathrm{d} \\ x_\mathrm{q} \\ x_\upgamma \end{bmatrix} = \begin{bmatrix}
            \cos(\varepsilon)  & \sin(\varepsilon) & 0 \\
            -\sin(\varepsilon) & \cos(\varepsilon) & 0 \\
            0                  & 0                 & 1
        \end{bmatrix} \begin{bmatrix} x_\upalpha \\ x_\upbeta \\ x_\upgamma \end{bmatrix}.
        \label{eq:Park_transform_dqgamma}
    \end{equation}
    \pause

    \begin{varblock}{Note on the $\upgamma$ component in $\mathrm{dq}$ coordinates}
        While the $\mathrm{dq}$ components of transformed currents / voltages / fluxes become constant in steady state, the $\upgamma$ / zero component typically exhibits harmonics (e.g., due to non-idealities in the machine, unbalanced loads, etc.) and thus does not become constant. Therefore, the $\upgamma$ component is often neglected in the control of three-phase machines, but it can be relevant for certain applications (e.g., fault detection).
    \end{varblock}
\end{frame}

\subsection{Power definitions}
\label{subsec:power_def}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% General setting %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{General setting: three-phase system with possible zero component}
    Consider arbitrary instantaneous three-phase voltage and current vectors
    \begin{equation}
        \bm{u}_{\mathrm{abc}}(t)=\begin{bmatrix} u_{\mathrm{a}}(t) & u_{\mathrm{b}}(t) & u_{\mathrm{c}}(t) \end{bmatrix}^\top,
        \qquad
        \bm{i}_{\mathrm{abc}}(t)=\begin{bmatrix} i_{\mathrm{a}}(t) & i_{\mathrm{b}}(t) & i_{\mathrm{c}}(t) \end{bmatrix}^\top,
    \end{equation}
    without assuming balanced sinusoidal operation or vanishing zero-sequence components. We define the \hl{instantaneous active power $p(t)$} as the inner product of the voltage and current vectors:
    \begin{equation}
        p(t)=\bm{u}_{\mathrm{abc}}^{\top}(t)\bm{i}_{\mathrm{abc}}(t) = u_{\mathrm{a}}(t)i_{\mathrm{a}}(t) + u_{\mathrm{b}}(t)i_{\mathrm{b}}(t) + u_{\mathrm{c}}(t)i_{\mathrm{c}}(t).
    \end{equation}
    Moreoever, we define the \hl{instantaneous reactive power vector $\bm{q}(t)$} as the cross product of the voltage and current vectors
    \begin{equation}
        \bm{q}_{\mathrm{abc}}(t)=\bm{u}_{\mathrm{abc}}(t)\times\bm{i}_{\mathrm{abc}}(t)=\begin{bmatrix}
            u_{\mathrm{b}}(t)i_{\mathrm{c}}(t)-u_{\mathrm{c}}(t)i_{\mathrm{b}}(t) \\
            u_{\mathrm{c}}(t)i_{\mathrm{a}}(t)-u_{\mathrm{a}}(t)i_{\mathrm{c}}(t) \\
            u_{\mathrm{a}}(t)i_{\mathrm{b}}(t)-u_{\mathrm{b}}(t)i_{\mathrm{a}}(t)
        \end{bmatrix}
    \end{equation}
    leading to \hl{(scalar) instantaneous reactive power $q(t)$} as the norm of $\bm{q}_{\mathrm{abc}}(t)$, i.e.,
    \begin{equation}
        q(t)=\lVert\bm{q}_{\mathrm{abc}}(t)\rVert.
    \end{equation}
\end{frame}

%% Clarke transformation %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Representation in $\upalpha\upbeta\upgamma$ coordinates}
    Using the amplitude-invariant Clarke transformation \eqref{eq:Clarke_transformation} produces
    \begin{equation}
        \bm{u}_{\mathrm{\upalpha\upbeta\upgamma}}(t)=\begin{bmatrix} u_{\mathrm{\upalpha}}(t) & u_{\mathrm{\upbeta}}(t) & u_{\mathrm{\upgamma}}(t) \end{bmatrix}^\top,
        \qquad
        \bm{i}_{\mathrm{\upalpha\upbeta\upgamma}}(t)=\begin{bmatrix} i_{\mathrm{\upalpha}}(t) & i_{\mathrm{\upbeta}}(t) & i_{\mathrm{\upgamma}}(t) \end{bmatrix}^\top,
    \end{equation}
    and the instantaneous active becomes
    \begin{equation}
        p(t)=\frac{3}{2}\bm{u}_{\upalpha\upbeta\upgamma}^{\top}(t)\bm{i}_{\upalpha\upbeta\upgamma}(t) = \frac{3}{2}\left(u_{\mathrm{\upalpha}}(t)i_{\mathrm{\upalpha}}(t) + u_{\mathrm{\upbeta}}(t)i_{\mathrm{\upbeta}}(t) + u_{\mathrm{\upgamma}}(t)i_{\mathrm{\upgamma}}(t)\right),
    \end{equation}
    while the instantaneous reactive power vector becomes
    \begin{equation}
        \bm{q}_{\upalpha\upbeta\upgamma}(t)=\frac{3}{2}\left(\bm{u}_{\mathrm{\upalpha\upbeta\upgamma}}(t)\times\bm{i}_{\mathrm{\upalpha\upbeta\upgamma}}(t)\right)=\frac{3}{2}
        \begin{bmatrix}
            u_{\mathrm{\upbeta}}(t)i_{\mathrm{\upgamma}}(t)-u_{\mathrm{\upgamma}}(t)i_{\mathrm{\upbeta}}(t)   \\
            u_{\mathrm{\upgamma}}(t)i_{\mathrm{\upalpha}}(t)-u_{\mathrm{\upalpha}}(t)i_{\mathrm{\upgamma}}(t) \\
            u_{\mathrm{\upalpha}}(t)i_{\mathrm{\upbeta}}(t)-u_{\mathrm{\upbeta}}(t)i_{\mathrm{\upalpha}}(t)
        \end{bmatrix}.
    \end{equation}
    Moreover, we define the \hl{instantaneous apparent power} as the norm of the instantaneous active and reactive powers (which applies in any coordinate system):
    \begin{equation}
        s(t)=\sqrt{p^2(t)+q^2(t)}.
    \end{equation}
\end{frame}

%% dqgamma coordinates %%

\begin{frame}
    \frametitle{Representation in $\mathrm{dq}\upgamma$ coordinates}
    The Park transformation \eqref{eq:Park_transform_dqgamma} delivers
    \begin{equation}
        \bm{u}_{\mathrm{dq}\upgamma}(t)=\begin{bmatrix} u_{\mathrm{d}}(t) & u_{\mathrm{q}}(t) & u_{\mathrm{\upgamma}}(t) \end{bmatrix}^\top,
        \qquad
        \bm{i}_{\mathrm{dq}\upgamma}(t)=\begin{bmatrix} i_{\mathrm{d}}(t) & i_{\mathrm{q}}(t) & i_{\mathrm{\upgamma}}(t) \end{bmatrix}^\top.
    \end{equation}
    Hence, the instantaneous active power becomes
    \begin{equation}
        p(t)=\frac{3}{2}\bm{u}_{\mathrm{dq}\upgamma}^{\top}(t)\bm{i}_{\mathrm{dq}\upgamma}(t) = \frac{3}{2}\left(u_{\mathrm{d}}(t)i_{\mathrm{d}}(t) + u_{\mathrm{q}}(t)i_{\mathrm{q}}(t) + u_{\mathrm{\upgamma}}(t)i_{\mathrm{\upgamma}}(t)\right),
    \end{equation}
    while the instantaneous reactive power vector becomes
    \begin{equation}
        \bm{q}_{\mathrm{dq}\upgamma}(t)=\frac{3}{2}\left(\bm{u}_{\mathrm{dq}\upgamma}(t)\times\bm{i}_{\mathrm{dq}\upgamma}(t)\right)=\frac{3}{2}
        \begin{bmatrix}
            u_{\mathrm{q}}(t)i_{\mathrm{\upgamma}}(t)-u_{\mathrm{\upgamma}}(t)i_{\mathrm{q}}(t) \\
            u_{\mathrm{\upgamma}}(t)i_{\mathrm{d}}(t)-u_{\mathrm{d}}(t)i_{\mathrm{\upgamma}}(t) \\
            u_{\mathrm{d}}(t)i_{\mathrm{q}}(t)-u_{\mathrm{q}}(t)i_{\mathrm{d}}(t)
        \end{bmatrix}.
    \end{equation}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% From instantaneous to average powers %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{From instantaneous quantities to average active and reactive powers}
    Let $T=\nicefrac{2\pi}{\omega}$ denote the \hl{fundamental period} of the system. Then, the classical \hl{average active power} follows directly from the instantaneous active power:
    \begin{equation}
        P(t)=\overline{p}(t)=\frac{1}{T}\int_{t-T}^{t} p(\tau)\,\mathrm{d}\tau.
    \end{equation}
    This relation is valid for all coordinate systems. Applying the same averaging to the instantaneous reactive power $q(t)$, however, does not in general recover the classical average reactive power $Q$:
    \begin{equation}
        \overline{q}(t)=\frac{1}{T}\int_{t-T}^{t} q(\tau)\,\mathrm{d}\tau \neq Q(t).
    \end{equation}

    \begin{varblock}{Difference between $Q$ and $\overline{q}$}
        The classical average reactive power $Q$ and the average of the instantaneous reactive power $\overline{q}$ are different concepts: $Q$ is a signed reactive power which differentiates between inductive and capacitive behavior, whereas $\overline{q}$ is a non-negative norm describing the average magnitude of the reactive power vector.
    \end{varblock}

\end{frame}

%% Balanced simplification %%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
    \frametitle{Balanced three-phase system without zero component}
    If the three-phase currents and/or voltages are zero-sequence free, i.e.,
    \begin{equation}
        x_{\upgamma}(t)=0 \quad \Leftrightarrow \quad x_{\mathrm{a}}(t)+x_{\mathrm{b}}(t)+x_{\mathrm{c}}(t)=0,
    \end{equation}
    the instantaneous active and reactive powers simplify to
    \begin{equation}
        \begin{split}
            p(t) & =\frac{3}{2}\Big(v_{\upalpha}(t)i_{\upalpha}(t)+v_{\upbeta}(t)i_{\upbeta}(t)\Big)
            =\frac{3}{2}\Big(v_{\mathrm{d}}(t)i_{\mathrm{d}}(t)+v_{\mathrm{q}}(t)i_{\mathrm{q}}(t)\Big), \\
            q(t) & =\frac{3}{2}\left|v_{\upalpha}(t)i_{\upbeta}(t)-v_{\upbeta}(t)i_{\upalpha}(t)\right|
            =\frac{3}{2}\left|v_{\mathrm{d}}(t)i_{\mathrm{q}}(t)-v_{\mathrm{q}}(t)i_{\mathrm{d}}(t)\right|.
        \end{split}
    \end{equation}
    Further assuming a steady-state balanced system, the instantaneous active and reactive powers become constant and equal to their average values, i.e.,
    \begin{equation}
        p(t)=P, \qquad q(t)=|Q|.
    \end{equation}
    Hence, only in this special case, the instantaneous reactive power $q(t)$ can be used to determine the magnitude of the average reactive power $Q$.
\end{frame}
